##### Install GKlib, METIS, ParMETIS in one shot #####
cd metis
make metis

##### If the above installation fails, then install them inviddually #####

# Install GKlib
cd GKlib 
make config prefix=$(pwd)
make install

# Install METIS
cd ../METIS 
make config prefix=$(pwd) gklib_path="$(pwd)/../GKlib"
make install

# Install ParMETIS
cd ../ParMETIS 
make config prefix=$(pwd) gklib_path="$(pwd)/../GKlib" metis_path="$(pwd)/../METIS"
make install

# Install preprocess
cd ../preprocessing
cmake -S . -B build 
cmake --build build -j4
